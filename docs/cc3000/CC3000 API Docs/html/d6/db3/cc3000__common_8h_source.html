72 //TX and RX buffer sizes - allow to receive and transmit maximum data at lengh 8.
73 #ifndef CC3000_TINY_DRIVER
74 #define TINY_CC3000_MAXIMAL_RX_SIZE 44
75 #define TINY_CC3000_MAXIMAL_TX_SIZE 59
76 #endif
77 //Defines for minimal and maximal RX buffer size. This size includes the spi header and hci header.
78 //The maximal buffer size derives from:
79 // MTU + HCI header + SPI header + sendto() agrs size
80 //The minimum buffer size derives from:
81 // HCI header + SPI header + max args size
82 //
83 // This buffer is used for receiving events and data.
84 // Note that the maximal buffer size as defined here is a maximal at all
85 // The packet can not be longer than MTU size and CC3000 does not support
86 // fragmentation.Note that the same buffer is used for reception of the data and
87 // events from CC3000. That is why the minium is defined.
88 // The calculation for the actual size of buffer for reception is:
89 // Given the maximal data size MAX_DATA that is expected to be received by application, the required buffer is:
90 // Using recv() or recvfrom():
91 // max(CC3000_MINIMAL_RX_SIZE, MAX_DATA + HEADERS_SIZE_DATA + fromlen + ucArgsize + 1)
92 // Using gethostbyname() with minimal buffer size will limit the host name returned to 99 bytes only.
93 // The 1 is used for the overrun detection
94 #define CC3000_MINIMAL_RX_SIZE (118 + 1)
95 #define CC3000_MAXIMAL_RX_SIZE (1519 + 1)
96 
97 //Defines for minimal and maximal TX buffer size.
98 // This buffer is used for sending events and data.
99 // Note that the maximal buffer size as defined here is a maximal at all
100 // The packet can not be longer than MTU size and CC3000 does not support
101 // fragmentation.Note that the same buffer is used for tranmission of the data and
102 // commnads. That is why the minium is defined.
103 // The calculation for the actual size of buffer for transmission is:
104 // Given the maximal data size MAX_DATA, the required buffer is:
105 // Using Sendto:
106 // max(CC3000_MINIMAL_TX_SIZE, MAX_DATA + SPI_HEADER_SIZE + SOCKET_SENDTO_PARAMS_LEN + SIMPLE_LINK_HCI_DATA_HEADER_SIZE + 1)
107 // Using Send():
108 // max(CC3000_MINIMAL_TX_SIZE, MAX_DATA + SPI_HEADER_SIZE + HCI_CMND_SEND_ARG_LENGTH + SIMPLE_LINK_HCI_DATA_HEADER_SIZE + 1)
109 // The 1 is used for the overrun detection
110 #define CC3000_MINIMAL_TX_SIZE (118 + 1)
111 #define CC3000_MAXIMAL_TX_SIZE (1519 + 1)
112 
113 //In order to determine your preferred buffer size, change CC3000_MAXIMAL_RX_SIZE and CC3000_MAXIMAL_TX_SIZE
114 //to a value between the minimal and maximal specified above.
115 // Note that the buyffers are allocated by SPI. In case you change the size of those buffers, you might need also
116 // to change the linker file, since for example on MSP430 FRAM devices the buffers are allocated in the FRAM section
117 // that is allocated manually and not by IDE. Thus its size changes...
118 
119 #ifndef CC3000_TINY_DRIVER
120 
121  #define CC3000_RX_BUFFER_SIZE (CC3000_MAXIMAL_RX_SIZE)
122  #define CC3000_TX_BUFFER_SIZE (CC3000_MAXIMAL_TX_SIZE)
123 
124 //if defined TINY DRIVER we use smaller rx and tx buffer in order to minimize RAM consumption
125 #else
126  #define CC3000_RX_BUFFER_SIZE (TINY_CC3000_MAXIMAL_RX_SIZE)
127  #define CC3000_TX_BUFFER_SIZE (TINY_CC3000_MAXIMAL_TX_SIZE)
128 
129 #endif

240 //This macro is used for copying 8 bit to stream while converting to little endian format.
241 #define UINT8_TO_STREAM(_p, _val) {*(_p)++ = (_val);}
242 //This macro is used for copying 16 bit to stream while converting to little endian format.
243 #define UINT16_TO_STREAM(_p, _u16) (UINT16_TO_STREAM_f(_p, _u16))
244 //This macro is used for copying 32 bit to stream while converting to little endian format.
245 #define UINT32_TO_STREAM(_p, _u32) (UINT32_TO_STREAM_f(_p, _u32))
246 //This macro is used for copying a specified value length bits (l) to stream while converting to little endian format.
247 #define ARRAY_TO_STREAM(p, a, l) {register short _i; for (_i = 0; _i < l; _i++) *(p)++ = ((unsigned char *) a)[_i];}
248 //This macro is used for copying received stream to 8 bit in little endian format.
249 #define STREAM_TO_UINT8(_p, _offset, _u8) {_u8 = (unsigned char)(*(_p + _offset));}
250 //This macro is used for copying received stream to 16 bit in little endian format.
251 #define STREAM_TO_UINT16(_p, _offset, _u16) {_u16 = STREAM_TO_UINT16_f(_p, _offset);}
252 //This macro is used for copying received stream to 32 bit in little endian format.
253 #define STREAM_TO_UINT32(_p, _offset, _u32) {_u32 = STREAM_TO_UINT32_f(_p, _offset);}
254 #define STREAM_TO_STREAM(p, a, l) {register short _i; for (_i = 0; _i < l; _i++) *(a)++ = (unsigned char) p[_i];}
